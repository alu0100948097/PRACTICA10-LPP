def each
    yield temp = @cabeza
    if tam!=0
        for i in (1 ..tam-1)
            yield temp = temp.prev
        end
    end
end

def introducir_en_cabeza!(valor)
    
    if(@tam==0)
        @cabeza=Node.new(valor,nil,nil)
        @cola= cabeza
        
        else
        @cabeza.next=Node.new(valor,cola,nil)
        @cabeza.next.prev= cabeza
        @cabeza= cabeza.next
        
    end
    
    @tam+=1
    
end

def quitar_cabeza!
    
    temp=@cabeza
    
    if(tam>0)
        @cabeza= cabeza.prev
        if cabeza.nil?
            @cola=nil
            else
            @cabeza.next=nil
        end
        temp.next=nil
        temp.prev=nil
        @tam-=1
    end
    
    temp
    
end
