def AIBC
    aibc=Array.new
    g.each do |x|
        almacen=Array.new
        vec_2=(1..x.length-1).to_a.zip(x.drop(1))
        vec_2.each{ |val|  almacen<<(((val[1]-x[0]+x[val[0]-1]-x[0])/2)*5).round(2)}
        aibc<<(almacen.reduce :+)
    end
    aibc
end
